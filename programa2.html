Programa #2: Conversión de Monedas

Elabore un programa en JAVA que a través de un Menú de opciones permita al usuario convertir de pesos a las siguientes monedas:

  • Dólar
  • Euro
  • Yen(Japonés)
  • Dólar de Hong Kong

Notas:

  • Las equivalencias deberán definirse dentro del programa.
  • Usar el cuadro de dialogo, showOptionDialog(), para seleccionar: Dólar, Euro, Yen, DolarHK)

Código:

import java.util.Scanner;
import java.text.DecimalFormat;
import javax.swing.JOptionPane;

public class tarea17 {
  public static void main(String[] args) {
    try (Scanner input = new Scanner(System.in)) {
    }

    DecimalFormat df = new DecimalFormat("#.00");
    int opc = 0;
    double peso = 0;
    double dolar = 0;
    double euro = 0;
    double yen = 0;
    double dolarh = 0;
    String mensaje = "";

    do {
      opc = Integer.parseInt(JOptionPane.showInputDialog(null,
          "Digite la conversión de moneda que desea hacer:\n1. Peso mexicano a dolar.
          \n2. Peso mexicano a euro.\n3. Peso mexicano a yen.\n4. Peso mexicano a dolar
          hongkonés.\n5. Salir"));
      switch (opc) {
        case 1:
          peso = Integer.parseInt(JOptionPane.showInputDialog(null, "Ingrese la cantidad de
          pesos"));

          if (peso > 0) {
            dolar = peso * 0.051;
            mensaje = "La cantidad de $" + peso + " MXN son $" + df.format(dolar);
          } else
            mensaje = "Ingrese valores positivos";
          JOptionPane.showMessageDialog(null, mensaje);
          break;

        case 2:
          peso = Integer.parseInt(JOptionPane.showInputDialog(null, "Ingrese la cantidad de
          pesos"));

          if (peso > 0) {
            euro = peso * 0.046;
            mensaje = "La cantidad de $" + peso + " MXN son €" + df.format(euro);
          } else
            mensaje = "Ingrese valores positivos";
          JOptionPane.showMessageDialog(null, mensaje);
          break;

        case 3:
          peso = Integer.parseInt(JOptionPane.showInputDialog(null, "Ingrese la cantidad de
          pesos"));

          if (peso > 0) {
            yen = peso * 5.42;
            mensaje = "La cantidad de $" + peso + " MXN son ¥" + df.format(yen);
          } else
            mensaje = "Ingrese valores positivos";
          JOptionPane.showMessageDialog(null, mensaje);
          break;

        case 4:
          peso = Integer.parseInt(JOptionPane.showInputDialog(null, "Ingrese la cantidad de
          pesos"));

          if (peso > 0) {
            dolarh = peso * 0.66;
            mensaje = "La cantidad de $" + peso + " MXN son $" + df.format(dolarh);
          } else
            mensaje = "Ingrese valores positivos";
          JOptionPane.showMessageDialog(null, mensaje);
          break;

        case 5:
          mensaje = "Has salido";
          JOptionPane.showMessageDialog(null, mensaje);
          break;
          
        default:
          mensaje = "Ingrese valores que estén dentro del menú.";
          JOptionPane.showMessageDialog(null, mensaje);
          break;
      }
    } while (opc != 5);
  }
}
