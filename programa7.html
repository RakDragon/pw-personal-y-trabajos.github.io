Programa #7: Multiplicación de Matrices

Elaborar un programa en JAVA que calcule la multiplicación de dos matrices que ingrese el usuario.
(Nota: Validar lo que sea necesario para poder realizar la multiplicación de matrices.)

Código:

import java.util.Scanner;

public class tarea22 {

    public static void main(String[] args) {

        int[][] matrix1;
        int[][] matrix2;
        int[][] matrixResult;
        int rows;
        int columns;
        int rows2;
        int columns2;
        try (Scanner input = new Scanner(System.in)) {
            System.out.println("Multiplicación de matrices");
            System.out.println("------------------------");

            System.out.print("Introduce el número de filas de la primera matriz: ");
            rows = input.nextInt();
            System.out.print("Introduce el número de columnas de la primera matriz: ");
            columns = input.nextInt();

            System.out.print("Introduce el número de filas de la segunda matriz: ");
            rows2 = input.nextInt();
            System.out.print("Introduce el número de columnas de la segunda matriz: ");
            columns2 = input.nextInt();

            if (columns != rows2) {
                System.out.println("La multiplicación de matrices no se puede realizar
                porque el número de columnas de la primera matriz no coincide con el número
                de filas de la segunda.");
            } else {

                matrix1 = new int[rows][columns];
                matrix2 = new int[rows2][columns2];
                matrixResult = new int[rows][columns2];

                System.out.println("Introduce los elementos de la primera matriz:");
                for (int i = 0; i < rows; i++) {
                    for (int j = 0; j < columns; j++) {
                        System.out.print("[" + (i) + "][" + (j) + "]: ");
                        matrix1[i][j] = input.nextInt();
                    }
                }

                System.out.println("Introduce los elementos de la segunda matriz:");
                for (int i = 0; i < rows2; i++) {
                    for (int j = 0; j < columns2; j++) {
                        System.out.print("[" + (i) + "][" + (j) + "]: ");
                        matrix2[i][j] = input.nextInt();
                    }
                }

                System.out.println("La matriz resultante de la multiplicación de las
                matrices anteriores es:");
                for (int i = 0; i < rows; i++) {
                    for (int j = 0; j < columns2; j++) {
                        for (int k = 0; k < columns; k++) {
                            matrixResult[i][j] += matrix1[i][k] * matrix2[k][j];
                        }
                        System.out.print(matrixResult[i][j] + " ");
                    }
                    System.out.println();
                }
            }
        }
    }
}
