Programa #3: Serie A

Construya un programa en JAVA que calcule e imprima la siguiente serie para los N primeros términos:

Código:

import javax.swing.JOptionPane;

public class tarea18 {

    public static void main (String[] args) {

        String entrada;
        int i,n,num;

        entrada = JOptionPane.showInputDialog("Ingrese el total de números en la serie: ");
        n = Integer.parseInt(entrada);
        String letrero;
        letrero = "Los números de la serie son: ";
        
        for(i = 1; i <= n; i++){
            num = (int) Math.pow (i,2*i-1);
            if (i==n){
                letrero = letrero + num + ".";
            }
            else{
                letrero = letrero + num + ", ";
            }
        }
        JOptionPane.showMessageDialog(null, "" + letrero);
    }
}
