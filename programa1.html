Programa #1: Conversión de Temperatura

Elabore un programa en JAVA que mediante un Menú de opciones permita calcular la conversión de escala de temperatura:

  • De grados Centígrados a grados Farenheit
  • De grados Farenheit a grados Centígrados

(Nota: Usar el cuadro de dialogo, showOptionDialog(), para seleccionar: °C a °F y otro °F a °C)

Código:

import javax.swing.JOptionPane;

public class tarea16 {
    public static void main(String[] args) {
        int opcion = JOptionPane.showOptionDialog(null, "Elija la opción de conversion de
        temperatura",
        "Conversión de temperatura", JOptionPane.YES_NO_OPTION,
        JOptionPane.INFORMATION_MESSAGE, null, new Object[]{"Centígrados a Fahrenheit",
        "Fahrenheit a centígrados"}, "Default");

        switch (opcion) {
            case 0:
                System.out.println("Conversión de centígrados a Fahrenheit");
                int grados = Integer.parseInt(JOptionPane.showInputDialog(null, "Ingrese la
                temperatura en centígrados"));
                int fahrenheit = (int) ((9.0 / 5.0) * grados + 32);
                JOptionPane.showMessageDialog(null, "La temperatura en Fahrenheit es: " +
                fahrenheit);
                break;
            case 1:
                System.out.println("Conversión de Fahrenheit a centígrados");
                int fahrenheit2 = Integer.parseInt(JOptionPane.showInputDialog(null,
                "Ingrese la temperatura en Fahrenheit"));
                int centigrados = (int) ((5.0 / 9.0) * (fahrenheit2 - 32));
                JOptionPane.showMessageDialog(null, "La temperatura en centígrados es: " +
                centigrados);
                break;
        }
    }
}
